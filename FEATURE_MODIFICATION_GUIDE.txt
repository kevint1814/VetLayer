   trajectory_bonus = 0.0
   if parsed_resume and parsed_resume.get("experience_trajectory"):
       traj = parsed_resume["experience_trajectory"]
       trajectory_score = traj.get("trajectory_score", 0) / 100.0
       gap_penalty = max(0, traj.get("gap_years", 0) * 2)  # 2 points per year gap
       
       # Trajectory adds 0-10% bonus
       trajectory_bonus = trajectory_score * 0.10 - (gap_penalty / 100.0)
   
   # Incorporate into final score
   overall_score = min(100, max(0, overall_score + trajectory_bonus * 10))
   
   # Log trajectory impact
   if trajectory_bonus != 0:
       logger.info(f"Trajectory adjustment: {trajectory_bonus:+.2f}")
